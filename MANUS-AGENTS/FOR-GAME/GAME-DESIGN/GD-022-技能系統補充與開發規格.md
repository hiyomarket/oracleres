## 三、覺醒系統後端流程

### 3.1 覺醒觸發條件判斷

```typescript
// server/skills.ts — 覺醒條件檢查函數
export function checkAwakeCondition(
  skill: AgentSkill,
  template: SkillTemplate,
  materials: AwakeMaterial[]
): { canAwake: boolean; reason?: string } {
  const currentTier = skill.awakeTier;

  if (currentTier === 0) {
    // 普通 → 大招：純使用次數，無素材消耗
    if (skill.useCount < 100) {
      return { canAwake: false, reason: `還需使用 ${100 - skill.useCount} 次` };
    }
    return { canAwake: true };
  }

  if (currentTier === 1) {
    // 大招 → 奧義：使用次數 + 屬性精華 ×10 + 50,000 遊戲幣
    if (skill.useCount < 300) {
      return { canAwake: false, reason: `還需使用 ${300 - skill.useCount} 次` };
    }
    const essenceKey = `${template.element}_essence`;
    const essence = materials.find(m => m.materialType === essenceKey);
    if (!essence || essence.quantity < 10) {
      return { canAwake: false, reason: `缺少 ${template.element} 屬性精華（需 10 個）` };
    }
    return { canAwake: true };
  }

  if (currentTier === 2) {
    // 奧義 → 神技：天命碎片 ×1 + 屬性精華 ×50 + 200,000 遊戲幣
    const fateShard = materials.find(m => m.materialType === "fate_shard");
    if (!fateShard || fateShard.quantity < 1) {
      return { canAwake: false, reason: "缺少天命碎片" };
    }
    const essenceKey = `${template.element}_essence`;
    const essence = materials.find(m => m.materialType === essenceKey);
    if (!essence || essence.quantity < 50) {
      return { canAwake: false, reason: `缺少 ${template.element} 屬性精華（需 50 個）` };
    }
    return { canAwake: true };
  }

  return { canAwake: false, reason: "已達最高覺醒階段" };
}
```

### 3.2 覺醒成功率與失敗懲罰

```typescript
const AWAKE_RATES = { 0: 0.80, 1: 0.60, 2: 0.40 };

export function resolveAwake(currentTier: number): {
  success: boolean;
  newTier: number;
  penaltyTier: number;
} {
  const rate = AWAKE_RATES[currentTier as keyof typeof AWAKE_RATES] ?? 0;
  const success = Math.random() < rate;

  if (success) {
    return { success: true, newTier: currentTier + 1, penaltyTier: currentTier + 1 };
  }

  // 失敗懲罰：普通→大招失敗只扣素材；大招→奧義失敗退回普通；奧義→神技失敗退回奧義
  const penaltyTier = currentTier === 1 ? 0 : currentTier;
  return { success: false, newTier: penaltyTier, penaltyTier };
}
```

## 四、Combo 系統在 Tick 引擎中的觸發機制

### 4.1 設計原則（放置遊戲適配）

放置遊戲不像即時戰鬥可以讓玩家手動選擇技能順序，因此 Combo 系統改為**策略預設**：玩家在策略台設定「技能優先序」，Tick 引擎按照優先序自動使用技能，若連續使用符合 Combo 條件的技能，自動觸發 Combo 加成。

### 4.2 Combo 觸發邏輯

```typescript
// 在 tickEngine.ts 的戰鬥結算中加入 Combo 追蹤
interface ComboBuildState {
  chain: string[];        // 本回合已使用的技能 ID 序列
  comboBonus: number;     // 累積 Combo 加成倍率
  lastElement: string;    // 上一個技能的屬性
}

const WUXING_SHENG = {
  wood: "fire", fire: "earth", earth: "metal", metal: "water", water: "wood"
};

const CROSS_COMBO_EFFECTS: Record<string, { bonus: number; effect: string }> = {
  "wood+fire": { bonus: 0.5, effect: "burn_amplify" },
  "fire+earth": { bonus: 0.2, effect: "stun_chance_up" },
  "earth+metal": { bonus: 0.0, effect: "armor_break" },
  "metal+wood": { bonus: 0.0, effect: "bleed_extend" },
  "water+fire": { bonus: 0.3, effect: "steam_damage" },
  "water+earth": { bonus: 0.0, effect: "slow_amplify" },
};

export function calculateComboBonus(
  skillSequence: SkillTemplate[],
  comboState: ComboBuildState
): { multiplier: number; effect: string | null } {
  if (skillSequence.length < 2) return { multiplier: 1.0, effect: null };

  // 五行協力 Combo（跨屬性）
  const pairKey = `${comboState.lastElement}+${skillSequence[0].element}`;
  const crossCombo = CROSS_COMBO_EFFECTS[pairKey];
  if (crossCombo) {
    return { multiplier: 1.0 + crossCombo.bonus, effect: crossCombo.effect };
  }

  // 同屬性連擊 Combo
  const chainLength = comboState.chain.filter(
    id => id.startsWith(`S_${skillSequence[0].element.slice(0, 2).toUpperCase()}`)
  ).length;

  if (chainLength >= 5) return { multiplier: 3.0, effect: "seal" };
  if (chainLength >= 4) return { multiplier: 2.0, effect: "weaken" };
  if (chainLength >= 3) return { multiplier: 1.5, effect: "stun" };

  return { multiplier: 1.0, effect: null };
}
```

### 4.3 技能共鳴（被動加成）計算

```typescript
export function calculateSkillResonance(installedSkills: AgentSkill[]): Record<string, number> {
  const elementCounts: Record<string, number> = {};
  for (const skill of installedSkills) {
    const el = skill.element;
    elementCounts[el] = (elementCounts[el] ?? 0) + 1;
  }

  const resonanceBonus: Record<string, number> = {};
  for (const [el, count] of Object.entries(elementCounts)) {
    if (count >= 7) resonanceBonus[el] = 0.40;
    else if (count >= 5) resonanceBonus[el] = 0.25;
    else if (count >= 3) resonanceBonus[el] = 0.10;
    else resonanceBonus[el] = 0;
  }
  return resonanceBonus;
}
```

## 五、技能與命格五行的連動加成

### 5.1 命格加成對技能的影響

玩家的命格五行比例（木/火/土/金/水，總和 100%）對對應屬性技能有額外加成，這是天命共振最核心的差異化設計。

```typescript
// 命格技能加成公式
export function calcNatalSkillBonus(
  natalStats: { wood: number; fire: number; earth: number; metal: number; water: number },
  skillElement: string
): number {
  const elementMap: Record<string, keyof typeof natalStats> = {
    wood: "wood", fire: "fire", earth: "earth", metal: "metal", water: "water"
  };
  const proportion = natalStats[elementMap[skillElement] ?? "wood"] / 100;

  // 命格比例 → 技能加成對照
  // 0~20%：無加成
  // 20~40%：+10%
  // 40~60%：+25%
  // 60~80%：+40%
  // 80~100%：+60%（純命格玩家的極致加成）
  if (proportion >= 0.80) return 0.60;
  if (proportion >= 0.60) return 0.40;
  if (proportion >= 0.40) return 0.25;
  if (proportion >= 0.20) return 0.10;
  return 0;
}
```

### 5.2 命格弱點設計（強迫補強）

命格比例低的屬性，對應技能的效果會有輕微懲罰，迫使玩家透過裝備或寵物補強弱點，而不是無腦堆主屬性。

| 命格比例 | 對應技能效果 |
|---------|------------|
| 0~10% | 技能效果 ×0.70（明顯弱點） |
| 10~20% | 技能效果 ×0.85（輕微弱點） |
| 20%+ | 正常效果 ×1.00 |

## 六、隱藏技能觸發條件的後端追蹤機制

### 6.1 追蹤器設計原則

每個隱藏技能對應一個或多個追蹤器（Tracker），Tick 引擎在每次結算後更新相關追蹤器的計數值，達到目標值時自動觸發技能習得。

### 6.2 追蹤器 ID 命名規則

```
{技能ID}_{追蹤類型}
例：
S_Wd051_kill_count          木屬性隱藏技能1：擊殺木屬性怪物計數
S_X032_low_battery_wins     跨屬性隱藏技能32：低電量勝利計數
S_M053_auction_trades       金屬性隱藏技能3：拍賣行交易次數
```

### 6.3 Tick 引擎中的追蹤器更新

```typescript
// 在 tickEngine.ts 的 resolveTick() 中加入
export async function updateHiddenSkillTrackers(
  agentId: number,
  event: TickEvent
): Promise<void> {
  const db = await getDb();
  const updates: Array<{ trackerId: string; increment: number }> = [];

  // 戰鬥事件追蹤
  if (event.type === "combat" && event.result === "win") {
    // 追蹤擊殺特定屬性怪物
    if (event.monsterElement) {
      updates.push({
        trackerId: `kill_${event.monsterElement}_count`,
        increment: 1
      });
    }
    // 追蹤特定地點戰鬥
    if (event.nodeRegion) {
      updates.push({
        trackerId: `battle_${event.nodeRegion}_count`,
        increment: 1
      });
    }
  }

  // 採集事件追蹤
  if (event.type === "gather") {
    updates.push({
      trackerId: `gather_${event.element}_count`,
      increment: event.quantity ?? 1
    });
  }

  // 批次更新追蹤器
  for (const { trackerId, increment } of updates) {
    await db.execute(sql`
      INSERT INTO hidden_skill_trackers (agent_id, tracker_id, current_value, target_value)
      VALUES (${agentId}, ${trackerId}, ${increment}, 999999)
      ON DUPLICATE KEY UPDATE current_value = current_value + ${increment}
    `);
  }

  // 檢查是否觸發隱藏技能
  await checkAndUnlockHiddenSkills(agentId, db);
}

async function checkAndUnlockHiddenSkills(agentId: number, db: Database): Promise<void> {
  const trackers = await db.select()
    .from(hiddenSkillTrackers)
    .where(and(
      eq(hiddenSkillTrackers.agentId, agentId),
      eq(hiddenSkillTrackers.isUnlocked, false)
    ));

  for (const tracker of trackers) {
    if (tracker.currentValue >= tracker.targetValue) {
      // 習得隱藏技能
      const skillId = tracker.trackerId.split("_")[0] + "_" + tracker.trackerId.split("_")[1];
      await db.insert(agentSkills).values({
        agentId,
        skillId,
        awakeTier: 0,
        useCount: 0,
        installedSlot: null
      }).onDuplicateKeyUpdate({ set: { awakeTier: 0 } });

      // 標記追蹤器已解鎖
      await db.update(hiddenSkillTrackers)
        .set({ isUnlocked: true, unlockedAt: new Date() })
        .where(eq(hiddenSkillTrackers.id, tracker.id));

      // 檢查全服首觸發
      const existing = await db.select()
        .from(globalFirstTriggers)
        .where(eq(globalFirstTriggers.skillId, skillId))
        .limit(1);

      if (existing.length === 0) {
        const agent = await db.select().from(gameAgents)
          .where(eq(gameAgents.id, agentId)).limit(1);
        await db.insert(globalFirstTriggers).values({
          skillId,
          firstAgentId: agentId,
          firstAgentName: agent[0]?.travelerName ?? "未知旅人"
        });
        // TODO: 觸發全服公告通知
      }
    }
  }
}
```
